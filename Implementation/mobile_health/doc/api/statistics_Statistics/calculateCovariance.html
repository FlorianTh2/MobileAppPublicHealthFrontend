calculateCovariance(Array x, Array y) {
  var sumOfX =
      x.reduce((previousValue, element) => previousValue + (element - mean(x)));
  var sumOfY =
      y.reduce((previousValue, element) => previousValue + (element - mean(y)));
  var covariance = (sumOfX * sumOfY * 1 / (x.length - 1));

  return covariance;
}
